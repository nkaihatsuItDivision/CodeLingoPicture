🚀 Thymeleafテンプレートでの実装例
<!-- student-list.html -->
<!DOCTYPE html>
<html xmlns:th="http://www.thymeleaf.org">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <link rel="stylesheet" th:href="@{/css/responsive-style.css}">
  <title>学生一覧</title>
</head>
<body>
  <div class="container">
    <h1>学生一覧表</h1>
    <table class="student-table">
      <thead>
        <tr>
          <th>名前</th>
          <th>年齢</th>
          <th>専攻</th>
        </tr>
      </thead>
      <tbody>
        <tr class="student-row" th:each="student : ${students}">
          <td class="student-data" th:text="${student.name}"></td>
          <td class="student-data" th:text="${student.age}"></td>
          <td class="student-data" th:text="${student.major}"></td>
        </tr>
      </tbody>
    </table>
  </div>
</body>
</html>
