☕ Week 5: Java配列処理
// 年齢が20歳の学生を検索
for (Student student : students) {
if (student.getAge() == 20) {
System.out.println(student.getName());
}
}

// 工学部の学生を検索
for (Student student : students) {
if (student.getDepartment().contains("工学")) {
results.add(student);
}
}

// 複雑な条件検索
for (Student student : students) {
if (student.getAge() >= 20 &&
student.getDepartment().contains("工学")) {
results.add(student);
}
}
VS
🗄️ Week 7.3: SQL処理
-- 年齢が20歳の学生を検索
SELECT * FROM student
WHERE age = 20;

-- 工学部の学生を検索
SELECT * FROM student
WHERE department LIKE '%工学%';

-- 複雑な条件検索
SELECT * FROM student
WHERE age >= 20
AND department LIKE '%工学%';
1
基本検索パターン
Java: if (student.getAge() == 20)
SQL: WHERE age = 20
2
部分一致検索
Java: .contains("工学")
SQL: LIKE '%工学%'
3
複合条件検索
Java: age >= 20 && dept.contains("工学")
SQL: age >= 20 AND dept LIKE '%工学%'
4
並び替え処理
Java: Collections.sort(students, comparator)
SQL: ORDER BY age ASC
5
件数制限処理
Java: for(...) { if (count < limit) break; }
SQL: LIMIT 3
6
範囲検索
Java: age > 20 && age < 25
SQL: age BETWEEN 20 AND 25
