function showAutoHideMessage() {
  const message = document.getElementById('autoHideDemo');
  message.style.display = 'flex';
  setTimeout(() => {
    message.style.display = 'none';
  }, 3000);
}

function toggleCard(element) {
  element.classList.toggle('clicked');
  if (element.classList.contains('clicked')) {
    element.innerHTML = '<h4>詳細情報</h4><p>GPA: 85点</p>';
  }
}

// リアルタイムでバリデーション実行
function validateName(input) {
  const value = input.value.trim();
  if (value.length < 2) {
    input.className = 'form-input error';
    document.getElementById('nameError').style.display = 'block';
  } else {
    input.className = 'form-input success';
    document.getElementById('nameSuccess').style.display = 'block';
  }
}
