SELECT
    sub.subject_name, sub.credits,
    COUNT(sc.id) as students_enrolled,
    ROUND(AVG(sc.score), 1) as avg_score,
    MAX(sc.score) as max_score, MIN(sc.score) as min_score,
    SUM(CASE WHEN sc.grade_letter = 'A' THEN 1 ELSE 0 END) as grade_a_count,
    ROUND(AVG(CASE WHEN sc.score >= 60 THEN 1.0 ELSE 0.0 END) * 100, 1) as pass_rate
FROM subject sub
LEFT JOIN score sc ON sub.subject_code = sc.subject_code
GROUP BY sub.subject_code, sub.subject_name, sub.credits
ORDER BY avg_score DESC;

SELECT s.name, sc.semester,
    ROUND(AVG(sc.score), 1) as semester_avg,
    LAG(ROUND(AVG(sc.score), 1), 1) OVER (
        PARTITION BY s.student_id ORDER BY sc.semester
    ) as previous_avg,
    ROUND(AVG(sc.score), 1) - LAG(ROUND(AVG(sc.score), 1), 1) OVER (
        PARTITION BY s.student_id ORDER BY sc.semester
    ) as improvement
FROM student s INNER JOIN score sc ON s.student_id = sc.student_id
GROUP BY s.student_id, s.name, sc.semester;
