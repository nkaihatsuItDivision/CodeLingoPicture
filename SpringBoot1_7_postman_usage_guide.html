自動テスト機能
レスポンスの内容を自動検証
JavaScriptでテストスクリプト作成
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response has name", function () {
    pm.expect(pm.response.json().data.name)
        .to.eql("田中太郎");
});
