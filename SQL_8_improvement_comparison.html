📊 データベース改善効果の比較

❌ 改善前（レッスン7.7使用）
学生テーブル:
CREATE TABLE student (
id INT AUTO_INCREMENT PRIMARY KEY,
student_id VARCHAR(10) NOT NULL UNIQUE,
name VARCHAR(100) NOT NULL,
age INT NOT NULL,
department VARCHAR(100) NOT NULL, ← 重複データ
email VARCHAR(255),
enrollment_date DATE,
status ENUM('active', 'inactive', 'graduated')
);
科目テーブル:
CREATE TABLE subject (
id INT AUTO_INCREMENT PRIMARY KEY,
subject_code VARCHAR(10) NOT NULL UNIQUE,
subject_name VARCHAR(100) NOT NULL,
credits INT NOT NULL,
department VARCHAR(100) ← 重複データ
);
🚨 問題となる操作例
学部名変更時：
-- 複数テーブル・複数行の更新が必要
UPDATE student SET department = '情報工学科'
WHERE department = '情報工学部';
UPDATE subject SET department = '情報工学科'
WHERE department = '情報工学部';
-- 問題：更新漏れや入力ミスのリスク
学部別統計時：
-- 学部名の表記ゆれで正確な集計ができない
SELECT department, COUNT(*)
FROM student
GROUP BY department;
-- 結果例：
-- 情報工学部: 5名
-- 情報工学科: 3名 ← 同じ学部なのに分離
-- じょうほう工学部: 1名 ← 入力ミス
✅ 改善後（正規化適用）
学部テーブル（新規作成）:
CREATE TABLE department (
department_code VARCHAR(10) PRIMARY KEY,
department_name VARCHAR(100) NOT NULL UNIQUE,
building VARCHAR(50),
phone VARCHAR(20),
established_year INT
);
学生テーブル（改善）:
CREATE TABLE student (
student_id VARCHAR(10) PRIMARY KEY,
name VARCHAR(100) NOT NULL,
age INT NOT NULL,
department_code VARCHAR(10) NOT NULL, ← 外部キー
email VARCHAR(255) UNIQUE,
enrollment_date DATE NOT NULL,
status ENUM('active', 'inactive', 'graduated'),
FOREIGN KEY (department_code) REFERENCES department(department_code)
);
科目テーブル（改善）:
CREATE TABLE subject (
subject_code VARCHAR(10) PRIMARY KEY,
subject_name VARCHAR(100) NOT NULL,
credits INT NOT NULL,
department_code VARCHAR(10), ← 外部キー
FOREIGN KEY (department_code) REFERENCES department(department_code)
);
🎯 改善された操作例
学部名変更時：
-- 1テーブル・1行のみ更新
UPDATE department
SET department_name = '情報工学科'
WHERE department_code = 'CS';
-- 結果：全ての関連データが自動的に最新情報を参照
学部別統計時：
-- 正確で詳細な集計が可能
SELECT
d.department_name,
d.building,
COUNT(s.student_id) as student_count,
AVG(s.age) as avg_age
FROM department d
LEFT JOIN student s ON d.department_code = s.department_code
WHERE s.status = 'active'
GROUP BY d.department_code, d.department_name, d.building
ORDER BY student_count DESC;
-- 結果：一意の学部名で正確な集計
